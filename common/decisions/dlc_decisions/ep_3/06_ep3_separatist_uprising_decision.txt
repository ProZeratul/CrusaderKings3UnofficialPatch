# Lead a Separatist Uprising
#############################################

separatist_uprising_decision = {
	picture = { reference = "gfx/interface/illustrations/decisions/decision_culture.dds" }
	desc = separatist_uprising_decision_desc
	decision_group_type = adventurer
	selection_tooltip = separatist_uprising_decision_tooltip
	ai_check_interval = 12
	sort_order = 350

	is_shown = {
		has_ep3_dlc_trigger = yes
		is_landless_adventurer = yes
		NOT = { has_trait = populist_leader }
		NOT = { has_variable = separatist_uprising_decision_var }
		domicile.domicile_location.duchy = { #Unop: Only show this decision if the duchy exists
			is_title_created = yes
		}
	}
	
	is_valid = {
		exists = location.county
		exists = domicile.domicile_location.duchy.holder.liege
		
		domicile.domicile_location.duchy.holder.liege = { save_temporary_scope_as = recipient } #Unop: For the ai_will_do block
		domicile.domicile_location.duchy.title_capital_county = { save_temporary_scope_as = unop_uprising_capital_county } #Unop: For the yml
		domicile.domicile_location.duchy.title_capital_county.culture = { save_temporary_scope_as = unop_uprising_culture } #Unop: For the yml

		custom_tooltip = {
			text = separatist_uprising_culture_county_tt
			domicile.domicile_location.duchy.title_capital_county.culture = root.culture
		}
		custom_tooltip = {
			text = separatist_uprising_capital_county_tt
			NOT = { domicile.domicile_location.county = domicile.domicile_location.duchy.holder.liege.capital_county }
		}
		NOT = {
			domicile.domicile_location.duchy.holder.culture = root.culture
		}
		prestige_level >= 3
	}

	is_valid_showing_failures_only = {
		NOR = {
			is_available = no
			has_trait = infirm
			is_incapable = yes
			is_independent_ruler = no
		}
	}
	
	cost = {
		prestige = 2000
		gold = 500
	}

	effect = {
		show_as_tooltip = {
			add_pressed_claim = domicile.domicile_location.duchy
		}
		custom_tooltip = separatist_uprising_decision_war_tt
		if = {
			limit = {
				OR = {
					can_set_relation_potential_lover_trigger = { CHARACTER = scope:recipient } #Unop: Use our new registered scope (domicile location based)
					can_set_relation_lover_trigger = { CHARACTER = scope:recipient } #Unop: Use our new registered scope (domicile location based)
				}
			}
			custom_tooltip = separatist_uprising_decision_pacifist_option_friend_lover_tt
		}
		else = {
			custom_tooltip = separatist_uprising_decision_pacifist_option_friend_tt
		}
		trigger_event = ep3_decisions_event.6001
	}

	ai_will_do = {
		base = 50
		modifier = { # More likely to do it if they can actually win
			add = 100
			root_military_strength_higher_than_recipient_value > recipient_strength_root_value
		}
		modifier = { # They should never take it if they don't have enough Prestige and a bit more
			add = -200
			prestige <= 300
			gold <= 150
		}
		modifier = { # Asen and Peter always try to do this
			add = 1000
			primary_title = title:d_laamp_asen
		}
	}
}
